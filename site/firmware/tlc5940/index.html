    Driver TLC5940¶

    Initialisation SPI¶

    spi_bus_config_t bus_tlc = {
    .miso_io_num   = -1,
    .mosi_io_num   = TLC_MOSI,    // GPIO23
    .sclk_io_num   = TLC_SCLK,    // GPIO18
    .max_transfer_sz = 128
};
spi_device_interface_config_t dev_tlc = {
    .clock_speed_hz = 5000000,    // 5 MHz
    .mode           = 0,           // CPOL=0, CPHA=0
    .spics_io_num   = -1,          // CS géré manuellement via XLAT
    .queue_size     = 7
};
spi_bus_initialize(SPI2_HOST, &bus_tlc, SPI_DMA_CH_AUTO);
spi_bus_add_device(SPI2_HOST, &dev_tlc, &spi_tlc);

    DMA activé

    SPI_DMA_CH_AUTO active le DMA automatiquement. Pour les trames de 456 octets, le DMA est indispensable pour ne pas bloquer le CPU pendant le transfert.

    GSCLK via LEDC¶

    ledc_timer_config_t timer = {
    .speed_mode      = LEDC_LOW_SPEED_MODE,
    .timer_num       = LEDC_TIMER_0,
    .duty_resolution = LEDC_TIMER_1_BIT,   // 1 bit → duty 50%
    .freq_hz         = 1000000,             // 1 MHz
    .clk_cfg         = LEDC_AUTO_CLK
};
ledc_channel_config_t ch = {
    .gpio_num = TLC_GSCLK,   // GPIO21
    .duty     = 1,
    .hpoint   = 0
};

    Calcul du refresh :

    GSCLK = 1 MHz  →  4096 cycles = 1 période PWM = 4.096 ms
Fréquence de refresh LEDs = 1 / 4.096ms ≈ 244 Hz

    Envoi Grayscale (update_tlc)¶

    La fonction update_tlc convertit le tableau uint16_t data_in[304] en trame SPI de 456 octets avec packing 12 bits :

    void update_tlc(const uint16_t *data_in) {
    uint8_t gs[24 * NUM_TLC];   // 456 octets
    memset(gs, 0, sizeof(gs));

    for (int i = 0; i < NUM_OUTPUTS; i++) {
        int physical = (NUM_OUTPUTS - 1) - i;  // Inversion chaîne
        uint16_t val = data_in[i] & 0x0FFF;    // Masque 12 bits
        int bit_pos  = physical * 12;
        int byte     = bit_pos / 8;

        if (bit_pos % 8 == 0) {
            gs[byte]     = val >> 4;
            gs[byte + 1] |= (val & 0x0F) << 4;
        } else {
            gs[byte]     |= val >> 8;
            gs[byte + 1]  = val & 0xFF;
        }
    }

    // Envoi SPI + XLAT
    spi_device_transmit(spi_tlc, &t);
    gpio_set_level(TLC_BLANK, 1);
    gpio_set_level(TLC_XLAT,  1); esp_rom_delay_us(1);
    gpio_set_level(TLC_XLAT,  0);
    gpio_set_level(TLC_BLANK, 0);
}

    Packing 12 bits¶

    Les 16 canaux × 12 bits d'un TLC5940 sont packés en 24 octets (192 bits) :

    Canal 0  : bits 191-180  (MSB en premier)
Canal 1  : bits 179-168
...
Canal 15 : bits  11-0

    L'inversion physical = (NUM_OUTPUTS-1) - i compense le fait que les données entrent par le dernier TLC de la chaîne en premier (les bits voyagent en sens inverse dans la cascade série).

    Dot Correction au démarrage¶

    void send_tlc_dc(void) {
    uint8_t dc[12 * NUM_TLC];   // 228 octets (6 bits × 16 ch × 19 TLC)
    memset(dc, 0xFF, sizeof(dc));  // Luminosité maximale partout

    gpio_set_level(TLC_VPRG, 1);  // Mode DC
    spi_device_transmit(spi_tlc, &t);
    // Impulsion XLAT
    gpio_set_level(TLC_VPRG, 0);  // Retour mode GS
}

    Ajustement de luminosité

    Pour réduire la consommation ou la chaleur de l'AMS1117, diminuer la valeur DC (ex: 0xA0 au lieu de 0xFF) réduit le courant de chaque canal sans changer la résistance Rref.
